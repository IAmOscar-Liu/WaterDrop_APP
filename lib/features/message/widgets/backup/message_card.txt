import 'package:flutter/material.dart';
import 'package:flutter_ad_ecommerce/models/message.dart';
import 'package:flutter_ad_ecommerce/utils/formatter.dart';

class MessageCard extends StatefulWidget {
  final Message message;
  final void Function()? onTap;
  final void Function()? onDelete;

  const MessageCard({
    super.key,
    required this.message,
    required this.onTap,
    this.onDelete,
  });

  @override
  State<MessageCard> createState() => _MessageCardState();
}

class _MessageCardState extends State<MessageCard> {
  bool _isDetailsExpanded = false;

  @override
  Widget build(BuildContext context) {
    return Container(
      margin: const EdgeInsets.symmetric(vertical: 4.0, horizontal: 16.0),
      decoration: BoxDecoration(
        color: widget.message.isRead
            ? const Color(0xFF1E2127)
            : const Color(0xFF2B303A),
        borderRadius: BorderRadius.circular(8),
        border: Border.all(color: const Color(0xFF393E46), width: 1),
      ),
      child: Material(
        color: Colors.transparent,
        child: InkWell(
          onTap: () {
            setState(() {
              _isDetailsExpanded = !_isDetailsExpanded;
            });
            if (!widget.message.isRead && widget.onTap != null) {
              widget.onTap!();
            }
          },
          borderRadius: BorderRadius.circular(8),
          child: Padding(
            padding: const EdgeInsets.all(16.0),
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                Row(
                  mainAxisAlignment: MainAxisAlignment.spaceBetween,
                  children: [
                    Expanded(
                      child: Text(
                        widget.message.title,
                        style: TextStyle(
                          color: widget.message.isRead
                              ? Colors.white70
                              : Colors.white,
                          fontSize: 16,
                          fontWeight: widget.message.isRead
                              ? FontWeight.normal
                              : FontWeight.bold,
                        ),
                        overflow: TextOverflow.ellipsis,
                      ),
                    ),
                    const SizedBox(width: 10),
                    Text(
                      Formatter.formatDateTime(widget.message.createdAt),
                      style: const TextStyle(
                        color: Colors.white70,
                        fontSize: 12,
                      ),
                    ),
                  ],
                ),
                const SizedBox(height: 8),
                Text(
                  widget.message.body,
                  style: const TextStyle(color: Colors.white70, fontSize: 14),
                  overflow: _isDetailsExpanded ? null : TextOverflow.ellipsis,
                ),
              ],
            ),
          ),
        ),
      ),
    );
  }
}
